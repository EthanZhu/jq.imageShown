How to use:
$('elem').imageShown({'options'});
setter:
$('elem').imageShown('option',{'options'});
$('elem').imageShown('option','option','value');
getter:
var option = $('elem').imageShown('option','option name');
$('elem').imageShown('option',{'options'});
